const express = require('express'); const sqlite3 = require('sqlite3').verbose(); const cors = require('cors'); const helmet = require('helmet'); const path = require('path'); const app = express(); const PORT = process.env.PORT || 3001; // 中间件 app.use(helmet()); app.use(cors({ origin: ['http://localhost:3000', 'http://127.0.0.1:3000', 'null'] // 允许本地文件访问 })); app.use(express.json()); app.use(express.static(path.join(__dirname, 'public'))); // 初始化数据库 const db = new sqlite3.Database('./personality_test.db', (err) => { if (err) { console.error('数据库连接失败:', err.message); } else { console.log('成功连接到SQLite数据库'); initDatabase(); } }); // 创建表 function initDatabase() { db.serialize(() => { // 创建测试结果表 db.run(`CREATE TABLE IF NOT EXISTS test_results ( id INTEGER PRIMARY KEY AUTOINCREMENT, timestamp DATETIME DEFAULT CURRENT_TIMESTAMP, personality_type TEXT NOT NULL, e_score INTEGER NOT NULL, n_score INTEGER NOT NULL, a_score INTEGER NOT NULL, c_score INTEGER NOT NULL, o_score INTEGER NOT NULL, user_info TEXT )`); // 创建维度统计视图 db.run(`CREATE VIEW IF NOT EXISTS dimension_stats AS SELECT AVG(e_score) as avg_e, AVG(n_score) as avg_n, AVG(a_score) as avg_a, AVG(c_score) as avg_c, AVG(o_score) as avg_o, COUNT(*) as total_tests FROM test_results`); }); } // API 路由 app.get('/', (req, res) => { res.json({ message: '校园人格测试后端服务', version: '1.0.0', endpoints: { save_result: '/api/save-result', get_stats: '/api/stats', export_data: '/api/export', personality_types: '/api/personality-types' } }); }); // 保存测试结果 app.post('/api/save-result', (req, res) => { const { personalityType, scores, userInfo } = req.body; if (!personalityType || !scores) { return res.status(400).json({ error: '缺少必要参数' }); } const query = `INSERT INTO test_results (personality_type, e_score, n_score, a_score, c_score, o_score, user_info) VALUES (?, ?, ?, ?, ?, ?, ?)`; db.run(query, [ personalityType, scores.E || 0, scores.N || 0, scores.A || 0, scores.C || 0, scores.O || 0, JSON.stringify(userInfo || {}) ], function(err) { if (err) { console.error('保存测试结果失败:', err.message); return res.status(500).json({ error: '保存失败' }); } res.json({ success: true, result_id: this.lastID, message: '测试结果已保存' }); }); }); // 获取统计数据 app.get('/api/stats', (req, res) => { const stats = {}; // 获取维度平均分 db.get('SELECT * FROM dimension_stats', (err, row) => { if (err) { return res.status(500).json({ error: '获取统计数据失败' }); } stats.dimensions = { E: Math.round((row.avg_e || 0) * 10) / 10, N: Math.round((row.avg_n || 0) * 10) / 10, A: Math.round((row.avg_a || 0) * 10) / 10, C: Math.round((row.avg_c || 0) * 10) / 10, O: Math.round((row.avg_o || 0) * 10) / 10 }; stats.total_tests = row.total_tests || 0; // 获取人格类型分布 db.all('SELECT personality_type, COUNT(*) as count FROM test_results GROUP BY personality_type', (err, rows) => { if (err) { return res.status(500).json({ error: '获取人格类型分布失败' }); } stats.personality_types = rows.map(row => ({ type: row.personality_type, count: row.count, percentage: Math.round((row.count / stats.total_tests) * 100) / 100 })); res.json(stats); }); }); }); // 导出数据 app.get('/api/export', (req, res) => { const format = req.query.format || 'json'; if (format === 'json') { exportJSON(res); } else if (format === 'csv') { exportCSV(res); } else { res.status(400).json({ error: '不支持的导出格式' }); } }); // 导出为JSON function exportJSON(res) { db.all('SELECT * FROM test_results ORDER BY timestamp DESC', (err, rows) => { if (err) { return res.status(500).json({ error: '导出数据失败' }); } const exportData = { export_time: new Date().toISOString(), total_records: rows.length, data: rows.map(row => ({ id: row.id, timestamp: row.timestamp, personality_type: row.personality_type, scores: { E: row.e_score, N: row.n_score, A: row.a_score, C: row.c_score, O: row.o_score }, user_info: JSON.parse(row.user_info || '{}') })) }; res.setHeader('Content-Disposition', 'attachment; filename=personality_test_data.json'); res.json(exportData); }); } // 导出为CSV function exportCSV(res) { db.all('SELECT * FROM test_results ORDER BY timestamp DESC', (err, rows) => { if (err) { return res.status(500).json({ error: '导出数据失败' }); } if (rows.length === 0) { return res.send('No data to export'); } // CSV 表头 const headers = ['ID', '时间戳', '人格类型', '外向性(E)', '神经质(N)', '宜人性(A)', '尽责性(C)', '开放性(O)', '用户信息']; // 转换数据为CSV格式 const csvContent = [ headers.join(','), ...rows.map(row => [ row.id, `"${row.timestamp}"`, `"${row.personality_type}"`, row.e_score, row.n_score, row.a_score, row.c_score, row.o_score, `"${JSON.stringify(JSON.parse(row.user_info || '{}')).replace(/"/g, '""')}"` ].join(',')) ].join('\n'); res.setHeader('Content-Disposition', 'attachment; filename=personality_test_data.csv'); res.setHeader('Content-Type', 'text/csv; charset=utf-8'); res.send(csvContent); }); } // 获取人格类型信息 app.get('/api/personality-types', (req, res) => { const personalityTypes = [ { id: 'campus_apollo', name: '校园阿波罗', dimensions: { E: '高', N: '低', A: '高', C: '中', O: '中' } }, { id: 'campus_athena', name: '校园雅典娜', dimensions: { E: '中', N: '低', A: '中', C: '高', O: '高' } }, { id: 'campus_dionysus', name: '校园狄俄尼索斯', dimensions: { E: '高', N: '中', A: '中', C: '低', O: '高' } }, { id: 'campus_hermes', name: '校园赫尔墨斯', dimensions: { E: '高', N: '低', A: '中', C: '中', O: '高' } }, { id: 'campus_artemis', name: '校园阿尔忒弥斯', dimensions: { E: '低', N: '低', A: '中', C: '高', O: '中' } }, { id: 'campus_zeus', name: '校园宙斯', dimensions: { E: '高', N: '中', A: '低', C: '高', O: '中' } }, { id: 'campus_demeter', name: '校园德墨忒尔', dimensions: { E: '中', N: '中', A: '高', C: '高', O: '低' } }, { id: 'campus_poseidon', name: '校园波塞冬', dimensions: { E: '中', N: '高', A: '中', C: '中', O: '高' } }, { id: 'campus_hades', name: '校园哈迪斯', dimensions: { E: '低', N: '高', A: '低', C: '高', O: '中' } }, { id: 'campus_hestia', name: '校园赫斯提亚', dimensions: { E: '低', N: '低', A: '高', C: '中', O: '低' } } ]; res.json(personalityTypes); }); // 健康检查 app.get('/health', (req, res) => { res.json({ status: 'ok', timestamp: new Date().toISOString() }); }); // 启动服务器 app.listen(PORT, () => { console.log(`服务器运行在 http://localhost:${PORT}`); console.log(`健康检查: http://localhost:${PORT}/health`); console.log(`API文档: http://localhost:${PORT}/`); }); // 优雅关闭 process.on('SIGINT', () => { db.close((err) => { if (err) { console.error('关闭数据库连接失败:', err.message); } else { console.log('数据库连接已关闭'); } process.exit(0); }); });
